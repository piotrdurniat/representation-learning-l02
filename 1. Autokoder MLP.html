from torch import nn

from .ae import Autoencoder


class MLPEncoder(nn.Module):

    def __init__(self, latent_dim: int):
        super().__init__()

        self._layers = nn.Sequential(
            nn.Flatten(),
            nn.Linear(1 * 28 * 28, 128),
            nn.ReLU(inplace=True),
            nn.Linear(128, 64),
            nn.ReLU(inplace=True),
            nn.Linear(64, latent_dim),
        )

    def forward(self, img):
        return self._layers(img)


class MLPDecoder(nn.Module):

    def __init__(self, latent_dim: int):
        super().__init__()

        self._layers = nn.Sequential(
            nn.Linear(latent_dim, 64),
            nn.ReLU(inplace=True),
            nn.Linear(64, 128),
            nn.ReLU(inplace=True),
            nn.Linear(128, 1 * 28 * 28),
            nn.Unflatten(dim=1, unflattened_size=(1, 28, 28)),
            nn.Sigmoid(),
        )

    def forward(self, z):
        return self._layers(z)


class MLPAutoencoder(Autoencoder):

    def __init__(self, latent_dim: int):
        super().__init__(
            encoder=MLPEncoder(latent_dim=latent_dim),
            decoder=MLPDecoder(latent_dim=latent_dim),
        )
        self.save_hyperparameters({"latent_dim": latent_dim})

Jak ju zostae wspomniane we wprowadzeniu, do deklaracji waciwego autokodera oraz uczenia tego modelu, zosta wykorzystany PyTorch-Lightning. Model deklarujemy jako klas dziedziczc po LightningModule, a nastpnie deklarujemy metody:

  • forward - czyli standardowe przejcie w przd modelu; w tym wypadku bdziemy gwnie zainteresowani reprezentacjami, std forward wywouje tylko modu kodera,
  • training_step - metoda obliczajca funkcj kosztu dla podanej paczki (batch) danych; w tym wypadku korzystamy z dodatkowej funkcji reconstruct, ktra wywouje zarwno forward, jak i przekazuje wyniki do dekodera; jako funkcj kosztu wykorzystujemy bd redniokwadratowy (MSE),
  • configure_optimizers - w tej metodzie tworzymy obiekt optymalizatora (tutaj: Adam) podajc rwnie wspczynnik uczenia (learning rate).

from lightning import LightningModule
import torch
from torch import nn
from torch.nn import functional as F


class Autoencoder(LightningModule):

    def __init__(self, encoder: nn.Module, decoder: nn.Module):
        super().__init__()

        self._encoder = encoder
        self._decoder = decoder

    def forward(self, x):
        return self._encoder(x)

    def reconstruct(self, x):
        return self._decoder(z=self(x))

    def training_step(self, batch, batch_size):
        x, _ = batch
        xr = self.reconstruct(x)

        loss = F.mse_loss(input=xr, target=x)
        self.log("train/mse", loss.detach().item())
        return loss

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=1e-3)

1.6. Odporno modelu na transformacje (symetrie)

Podczas ostatniego wykadu (Wykad 2) omwione zostay dwa podejcia do opisywania struktury danych, mianowicie: symetrie oraz koncepcja separacji skali. Dobry model powinien by odporny na transformacje wynikajce z akcji grupy symetrii zwizanej z konkretnym typem danych.

W przypadku obrazw, ktre tutaj rozwaamy na przykadzie zbioru MNIST, jedn z grup symetrii jest grupa translacji (przesuni). Oznacza to, e niezalenie pooenia danego obiektu (tutaj: cyfry) na obrazie, model powinien uzyskiwa tak sam reprezentacj (lub bardzo podobn).

Zbadamy teraz jaka jest odporno autokodera MLP na przesunicia cyfr na obrazie.

Zadanie 1.1 (2 pkt)

Zaimplementuj funkcj shift, ktra przesunie kady piksel na obrazie wejciowym img (rozmiaru 28 x 28) o dx pikseli w prawo oraz dy pikseli w d. Funkcja powinna obsugiwa:

  • ujemne wartoci dla tych parametrw przesuni, tzn. dx < 0 (w lewo) oraz dy < 0 (do gry),
  • "zawijanie" zawartoci, np. jeli piksel przy przesuniciu w prawo wyszedby poza widoczny obszar, powinien zosta "zawinity" i pojawi si z lewej strony obrazu.

import torch


def shift(img: torch.Tensor, dx: int, dy: int) -> torch.Tensor:
    return torch.roll(img, shifts=(dy, dx), dims=(-2, -1))

Zadanie 1.2 (0.5 pkt)

Czy autokoder MLP jest odporny na przesunicia obiektw? Z czego to moe wynika?

Student's answer Score: 0.5 / 0.5 (Top)

Autoenkoder MLP nie jest odporny na przesunicia obiektw. Wynika to z faktu, e sie neuronowa nie posiada wiedzy o strukturze obrazu, a jedynie o wartociach pikseli. W zwizku z tym, przesunicie obiektu na obrazie powoduje zmian wartoci pikseli, co wpywa na reprezentacj obiektu w przestrzeni ukrytej.
